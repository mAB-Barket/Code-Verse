Project 1: Console Calculator

This project brings together variables, operators, control flow, functions, and input validation. The calculator performs basic arithmetic, handles division-by-zero gracefully, and loops so the user can perform multiple calculations without restarting the program.

Features

  • Addition, subtraction, multiplication, and division
  • Input validation – rejects non-numeric input
  • Division-by-zero protection
  • Loop for continuous use until the user decides to quit

Full Code

#include <iostream>
#include <limits>
using namespace std;

double calculate(double a, double b, char op) {
    switch (op) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/':
            if (b == 0) {
                cout << "Error: Division by zero!" << endl;
                return 0;
            }
            return a / b;
        default:
            cout << "Error: Invalid operator!" << endl;
            return 0;
    }
}

bool getNumber(const string& prompt, double& result) {
    cout << prompt;
    if (!(cin >> result)) {
        cin.clear();
        cin.ignore(numeric_limits<streamsize>::max(), '\n');
        cout << "Invalid input. Please enter a number." << endl;
        return false;
    }
    return true;
}

int main() {
    cout << "===== Console Calculator =====" << endl;
    char choice = 'y';

    while (choice == 'y' || choice == 'Y') {
        double num1, num2;
        char op;

        if (!getNumber("Enter first number: ", num1)) continue;

        cout << "Enter operator (+, -, *, /): ";
        cin >> op;

        if (!getNumber("Enter second number: ", num2)) continue;

        double result = calculate(num1, num2, op);
        cout << num1 << " " << op << " " << num2
             << " = " << result << endl;

        cout << "\nAnother calculation? (y/n): ";
        cin >> choice;
    }

    cout << "Goodbye!" << endl;
    return 0;
}

Project 2: Number Guessing Game

This project reinforces random numbers, loops, conditionals, and user input. The program picks a random number and gives the player hints (“Too high” / “Too low”) until they guess correctly, tracking the number of attempts.

Features

  • Random number generation with <cstdlib> and <ctime>
  • Configurable difficulty (range of numbers)
  • Attempt counter and final score
  • Input validation

Full Code

#include <iostream>
#include <cstdlib>
#include <ctime>
#include <limits>
using namespace std;

int main() {
    srand(static_cast<unsigned>(time(nullptr)));

    cout << "===== Number Guessing Game =====" << endl;
    cout << "Select difficulty:" << endl;
    cout << "1. Easy   (1-50)" << endl;
    cout << "2. Medium (1-100)" << endl;
    cout << "3. Hard   (1-500)" << endl;
    cout << "Choice: ";

    int difficulty;
    cin >> difficulty;

    int maxRange;
    switch (difficulty) {
        case 1:  maxRange = 50;  break;
        case 3:  maxRange = 500; break;
        default: maxRange = 100; break;
    }

    int secretNumber = rand() % maxRange + 1;
    int guess = 0;
    int attempts = 0;

    cout << "\nI'm thinking of a number between 1 and "
         << maxRange << "." << endl;

    while (guess != secretNumber) {
        cout << "Enter your guess: ";
        if (!(cin >> guess)) {
            cin.clear();
            cin.ignore(numeric_limits<streamsize>::max(), '\n');
            cout << "Please enter a valid number." << endl;
            continue;
        }
        attempts++;

        if (guess < secretNumber) {
            cout << "Too low! Try again." << endl;
        } else if (guess > secretNumber) {
            cout << "Too high! Try again." << endl;
        } else {
            cout << "\nCongratulations! You guessed it in "
                 << attempts << " attempt"
                 << (attempts == 1 ? "" : "s") << "!" << endl;
        }
    }

    if (attempts <= 5) {
        cout << "Rating: EXCELLENT!" << endl;
    } else if (attempts <= 10) {
        cout << "Rating: Good job!" << endl;
    } else {
        cout << "Rating: Keep practising!" << endl;
    }

    return 0;
}

Suggested Extensions

Project 3: Student Grade Manager

This project combines structs/classes, vectors, file I/O, and algorithms to build a small student records system. Users can add students, view all records, calculate GPAs, and save/load data to a file.

Features

  • Student struct with name, ID, and vector of grades
  • Add, view, and search students
  • GPA calculation per student and class average
  • Save records to a text file and load on startup

Full Code

#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <numeric>
#include <iomanip>
#include <algorithm>
using namespace std;

struct Student {
    string name;
    int id;
    vector<double> grades;

    double gpa() const {
        if (grades.empty()) return 0.0;
        double sum = accumulate(grades.begin(), grades.end(), 0.0);
        return sum / grades.size();
    }

    string letterGrade() const {
        double g = gpa();
        if (g >= 90) return "A";
        if (g >= 80) return "B";
        if (g >= 70) return "C";
        if (g >= 60) return "D";
        return "F";
    }
};

void saveToFile(const vector<Student>& students,
                const string& filename) {
    ofstream file(filename);
    if (!file) {
        cerr << "Error: Cannot open file for writing." << endl;
        return;
    }
    for (const auto& s : students) {
        file << s.id << "|" << s.name;
        for (double g : s.grades) {
            file << "|" << g;
        }
        file << "\n";
    }
    cout << "Data saved to " << filename << endl;
}

vector<Student> loadFromFile(const string& filename) {
    vector<Student> students;
    ifstream file(filename);
    if (!file) return students;

    string line;
    while (getline(file, line)) {
        Student s;
        size_t pos = line.find('|');
        s.id = stoi(line.substr(0, pos));
        line = line.substr(pos + 1);

        pos = line.find('|');
        if (pos == string::npos) {
            s.name = line;
        } else {
            s.name = line.substr(0, pos);
            line = line.substr(pos + 1);

            while (!line.empty()) {
                pos = line.find('|');
                if (pos == string::npos) {
                    s.grades.push_back(stod(line));
                    break;
                }
                s.grades.push_back(stod(line.substr(0, pos)));
                line = line.substr(pos + 1);
            }
        }
        students.push_back(s);
    }
    return students;
}

void displayStudents(const vector<Student>& students) {
    if (students.empty()) {
        cout << "No students in the system." << endl;
        return;
    }
    cout << left << setw(8) << "ID"
         << setw(20) << "Name"
         << setw(10) << "GPA"
         << "Grade" << endl;
    cout << string(45, '-') << endl;

    for (const auto& s : students) {
        cout << left << setw(8) << s.id
             << setw(20) << s.name
             << fixed << setprecision(2)
             << setw(10) << s.gpa()
             << s.letterGrade() << endl;
    }
}

void addStudent(vector<Student>& students) {
    Student s;
    cout << "Enter student ID: ";
    cin >> s.id;
    cin.ignore();
    cout << "Enter student name: ";
    getline(cin, s.name);

    int numGrades;
    cout << "How many grades? ";
    cin >> numGrades;

    for (int i = 0; i < numGrades; i++) {
        double g;
        cout << "  Grade " << (i + 1) << ": ";
        cin >> g;
        s.grades.push_back(g);
    }
    students.push_back(s);
    cout << "Student added successfully!" << endl;
}

void searchStudent(const vector<Student>& students) {
    cout << "Enter student name to search: ";
    string query;
    cin.ignore();
    getline(cin, query);

    bool found = false;
    for (const auto& s : students) {
        if (s.name.find(query) != string::npos) {
            cout << "ID: " << s.id << "  Name: " << s.name
                 << "  GPA: " << fixed << setprecision(2)
                 << s.gpa() << "  Grade: "
                 << s.letterGrade() << endl;
            found = true;
        }
    }
    if (!found) cout << "No matching student found." << endl;
}

int main() {
    const string filename = "students.txt";
    vector<Student> students = loadFromFile(filename);
    cout << "===== Student Grade Manager =====" << endl;

    int choice;
    do {
        cout << "\n1. Add Student\n2. View All Students"
             << "\n3. Search Student\n4. Class Average"
             << "\n5. Save & Quit\nChoice: ";
        cin >> choice;

        switch (choice) {
            case 1: addStudent(students); break;
            case 2: displayStudents(students); break;
            case 3: searchStudent(students); break;
            case 4: {
                if (students.empty()) {
                    cout << "No students." << endl;
                } else {
                    double total = 0;
                    for (const auto& s : students)
                        total += s.gpa();
                    cout << "Class average GPA: " << fixed
                         << setprecision(2)
                         << total / students.size() << endl;
                }
                break;
            }
            case 5:
                saveToFile(students, filename);
                cout << "Goodbye!" << endl;
                break;
            default:
                cout << "Invalid choice." << endl;
        }
    } while (choice != 5);

    return 0;
}

Project 4: Text-Based Adventure Game

This project showcases OOP, inheritance, polymorphism, and choice-based navigation. The player explores rooms, collects items, and makes decisions that affect the outcome of the game.

Features

  • Room base class with virtual methods for description and actions
  • Multiple room types using inheritance
  • Player inventory system
  • Choice-driven navigation between rooms

Full Code

#include <iostream>
#include <vector>
#include <string>
#include <memory>
#include <map>
using namespace std;

// ---- Player ----
class Player {
    string name;
    int health;
    vector<string> inventory;
public:
    Player(const string& n) : name(n), health(100) {}

    const string& getName() const { return name; }
    int getHealth() const { return health; }

    void takeDamage(int dmg) {
        health -= dmg;
        if (health < 0) health = 0;
        cout << name << " took " << dmg
             << " damage! HP: " << health << endl;
    }

    void heal(int amt) {
        health += amt;
        if (health > 100) health = 100;
        cout << name << " healed " << amt
             << " HP! HP: " << health << endl;
    }

    void addItem(const string& item) {
        inventory.push_back(item);
        cout << "Picked up: " << item << endl;
    }

    bool hasItem(const string& item) const {
        for (const auto& i : inventory)
            if (i == item) return true;
        return false;
    }

    void showInventory() const {
        cout << "Inventory: ";
        if (inventory.empty()) { cout << "(empty)" << endl; return; }
        for (size_t i = 0; i < inventory.size(); i++) {
            if (i > 0) cout << ", ";
            cout << inventory[i];
        }
        cout << endl;
    }

    bool isAlive() const { return health > 0; }
};

// ---- Room (base) ----
class Room {
protected:
    string name;
    string description;
public:
    Room(const string& n, const string& desc)
        : name(n), description(desc) {}
    virtual ~Room() = default;

    virtual void enter(Player& player) {
        cout << "\n--- " << name << " ---" << endl;
        cout << description << endl;
    }

    virtual int getChoices() { return 2; }
    const string& getName() const { return name; }
};

// ---- Derived rooms ----
class ForestRoom : public Room {
public:
    ForestRoom()
        : Room("Dark Forest",
               "Tall trees surround you. A faint path leads north.\n"
               "You hear rustling in the bushes.") {}

    void enter(Player& p) override {
        Room::enter(p);
        cout << "1. Follow the path north\n"
             << "2. Investigate the bushes" << endl;
    }
};

class CaveRoom : public Room {
public:
    CaveRoom()
        : Room("Mysterious Cave",
               "The cave is cold and damp. You see a glowing crystal\n"
               "on a pedestal and a dark tunnel deeper inside.") {}

    void enter(Player& p) override {
        Room::enter(p);
        cout << "1. Take the crystal\n"
             << "2. Enter the dark tunnel" << endl;
    }
};

class TreasureRoom : public Room {
public:
    TreasureRoom()
        : Room("Treasure Chamber",
               "Gold coins and jewels fill the room!\n"
               "A locked chest sits in the centre.") {}

    void enter(Player& p) override {
        Room::enter(p);
        if (p.hasItem("Crystal Key")) {
            cout << "You use the Crystal Key to open the chest!\n"
                 << "Inside you find the Legendary Sword!" << endl;
            p.addItem("Legendary Sword");
        } else {
            cout << "The chest is locked. You need a key." << endl;
        }
        cout << "1. Go back\n2. Search the room" << endl;
    }
};

class DragonRoom : public Room {
public:
    DragonRoom()
        : Room("Dragon's Lair",
               "A massive dragon blocks your path!\n"
               "Its eyes glow red as it stares you down.") {}

    void enter(Player& p) override {
        Room::enter(p);
        if (p.hasItem("Legendary Sword")) {
            cout << "You raise the Legendary Sword! The dragon retreats!\n"
                 << "*** YOU WIN! Congratulations, " << p.getName()
                 << "! ***" << endl;
        } else {
            cout << "Without a weapon, the dragon attacks!" << endl;
            p.takeDamage(50);
        }
        cout << "1. Go back\n2. Stand your ground" << endl;
    }
};

int main() {
    cout << "===== Text Adventure =====" << endl;
    cout << "Enter your name: ";
    string name;
    getline(cin, name);
    Player player(name);

    // Create rooms
    vector<unique_ptr<Room>> rooms;
    rooms.push_back(make_unique<ForestRoom>());   // 0
    rooms.push_back(make_unique<CaveRoom>());     // 1
    rooms.push_back(make_unique<TreasureRoom>()); // 2
    rooms.push_back(make_unique<DragonRoom>());   // 3

    // Room connections: room index -> {choice1 destination, choice2}
    map<int, pair<int, int>> connections = {
        {0, {1, 2}}, // Forest -> Cave or Treasure
        {1, {0, 3}}, // Cave -> Forest or Dragon
        {2, {0, 3}}, // Treasure -> Forest or Dragon
        {3, {1, 2}}  // Dragon -> Cave or Treasure
    };

    int currentRoom = 0;

    while (player.isAlive()) {
        rooms[currentRoom]->enter(player);
        player.showInventory();

        // Pick up Crystal Key in cave
        if (currentRoom == 1) {
            player.addItem("Crystal Key");
        }

        cout << "Your choice: ";
        int choice;
        cin >> choice;

        if (choice == 1)
            currentRoom = connections[currentRoom].first;
        else
            currentRoom = connections[currentRoom].second;

        if (player.hasItem("Legendary Sword")
            && currentRoom == 3) {
            rooms[currentRoom]->enter(player);
            break;
        }
    }

    if (!player.isAlive()) {
        cout << "\nGame Over. " << player.getName()
             << " has fallen." << endl;
    }

    return 0;
}

Suggested Extensions

Project 5: Contact Book

This project applies CRUD operations, file persistence, string manipulation, and searching. The user can create, read, update, and delete contacts, with all data persisted to a file between sessions.

Features

  • Add, view, update, and delete contacts
  • Search by name (case-insensitive partial match)
  • Persistent storage using a text file
  • Input validation and clean menu system

Full Code

#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <algorithm>
#include <cctype>
using namespace std;

struct Contact {
    string name;
    string phone;
    string email;
};

string toLower(string s) {
    transform(s.begin(), s.end(), s.begin(),
              [](unsigned char c) { return tolower(c); });
    return s;
}

class ContactBook {
    vector<Contact> contacts;
    string filename;

    void save() {
        ofstream file(filename);
        for (const auto& c : contacts) {
            file << c.name << "|" << c.phone
                 << "|" << c.email << "\n";
        }
    }

    void load() {
        ifstream file(filename);
        string line;
        while (getline(file, line)) {
            Contact c;
            size_t p1 = line.find('|');
            size_t p2 = line.find('|', p1 + 1);
            if (p1 == string::npos || p2 == string::npos)
                continue;
            c.name  = line.substr(0, p1);
            c.phone = line.substr(p1 + 1, p2 - p1 - 1);
            c.email = line.substr(p2 + 1);
            contacts.push_back(c);
        }
    }

public:
    ContactBook(const string& file) : filename(file) { load(); }

    void addContact() {
        Contact c;
        cin.ignore();
        cout << "Name:  ";  getline(cin, c.name);
        cout << "Phone: ";  getline(cin, c.phone);
        cout << "Email: ";  getline(cin, c.email);
        contacts.push_back(c);
        save();
        cout << "Contact added!" << endl;
    }

    void viewAll() const {
        if (contacts.empty()) {
            cout << "No contacts." << endl;
            return;
        }
        for (size_t i = 0; i < contacts.size(); i++) {
            cout << "[" << (i + 1) << "] "
                 << contacts[i].name << " | "
                 << contacts[i].phone << " | "
                 << contacts[i].email << endl;
        }
    }

    void search() const {
        cin.ignore();
        cout << "Search name: ";
        string query;
        getline(cin, query);
        string lowerQuery = toLower(query);
        bool found = false;

        for (const auto& c : contacts) {
            if (toLower(c.name).find(lowerQuery) != string::npos) {
                cout << c.name << " | " << c.phone
                     << " | " << c.email << endl;
                found = true;
            }
        }
        if (!found) cout << "No matches found." << endl;
    }

    void updateContact() {
        viewAll();
        cout << "Enter contact number to update: ";
        int idx;
        cin >> idx;
        if (idx < 1 || idx > static_cast<int>(contacts.size())) {
            cout << "Invalid number." << endl;
            return;
        }
        Contact& c = contacts[idx - 1];
        cin.ignore();
        cout << "New name  (" << c.name << "): ";
        string input;
        getline(cin, input);
        if (!input.empty()) c.name = input;

        cout << "New phone (" << c.phone << "): ";
        getline(cin, input);
        if (!input.empty()) c.phone = input;

        cout << "New email (" << c.email << "): ";
        getline(cin, input);
        if (!input.empty()) c.email = input;

        save();
        cout << "Contact updated!" << endl;
    }

    void deleteContact() {
        viewAll();
        cout << "Enter contact number to delete: ";
        int idx;
        cin >> idx;
        if (idx < 1 || idx > static_cast<int>(contacts.size())) {
            cout << "Invalid number." << endl;
            return;
        }
        cout << "Delete " << contacts[idx - 1].name << "? (y/n): ";
        char confirm;
        cin >> confirm;
        if (confirm == 'y' || confirm == 'Y') {
            contacts.erase(contacts.begin() + (idx - 1));
            save();
            cout << "Contact deleted." << endl;
        }
    }
};

int main() {
    ContactBook book("contacts.txt");
    cout << "===== Contact Book =====" << endl;
    int choice;

    do {
        cout << "\n1. Add Contact\n2. View All\n3. Search"
             << "\n4. Update Contact\n5. Delete Contact"
             << "\n6. Quit\nChoice: ";
        cin >> choice;

        switch (choice) {
            case 1: book.addContact();    break;
            case 2: book.viewAll();       break;
            case 3: book.search();        break;
            case 4: book.updateContact(); break;
            case 5: book.deleteContact(); break;
            case 6: cout << "Goodbye!" << endl; break;
            default: cout << "Invalid choice." << endl;
        }
    } while (choice != 6);

    return 0;
}

  • Add contact groups or categories.
  • Implement sorting (alphabetical, by date added).
  • Export contacts to JSON or CSV.
  • Add favourites and speed-dial numbers.

Challenge 2: Palindrome Checker

Write a function that checks if a string reads the same forwards and backwards, ignoring case and non-alphanumeric characters.

#include <iostream>
#include <string>
#include <algorithm>
#include <cctype>
using namespace std;

bool isPalindrome(const string& input) {
    string cleaned;
    for (char c : input) {
        if (isalnum(static_cast<unsigned char>(c)))
            cleaned += tolower(static_cast<unsigned char>(c));
    }
    string reversed = cleaned;
    reverse(reversed.begin(), reversed.end());
    return cleaned == reversed;
}

int main() {
    string tests[] = {
        "racecar", "A man, a plan, a canal: Panama",
        "hello", "Was it a car or a cat I saw?"
    };
    for (const auto& t : tests) {
        cout << "\"" << t << "\" -> "
             << (isPalindrome(t) ? "Palindrome" : "Not a palindrome")
             << endl;
    }
    return 0;
}

Challenge 3: Matrix Transposition

Given an M×N matrix, produce the N×M transpose.

#include <iostream>
#include <vector>
using namespace std;

using Matrix = vector<vector<int>>;

Matrix transpose(const Matrix& mat) {
    if (mat.empty()) return {};
    int rows = mat.size();
    int cols = mat[0].size();
    Matrix result(cols, vector<int>(rows));
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            result[j][i] = mat[i][j];
    return result;
}

void printMatrix(const Matrix& mat) {
    for (const auto& row : mat) {
        for (int val : row)
            cout << val << "\t";
        cout << endl;
    }
}

int main() {
    Matrix mat = {
        {1, 2, 3},
        {4, 5, 6}
    };

    cout << "Original:" << endl;
    printMatrix(mat);

    Matrix t = transpose(mat);
    cout << "\nTransposed:" << endl;
    printMatrix(t);

    return 0;
}

Challenge 4: Linked List Implementation

Implement a singly linked list with insert, remove, search, and display operations using dynamic memory and pointers.

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
    Node(int val) : data(val), next(nullptr) {}
};

class LinkedList {
    Node* head;
public:
    LinkedList() : head(nullptr) {}

    ~LinkedList() {
        while (head) {
            Node* temp = head;
            head = head->next;
            delete temp;
        }
    }

    void insertFront(int val) {
        Node* node = new Node(val);
        node->next = head;
        head = node;
    }

    void insertBack(int val) {
        Node* node = new Node(val);
        if (!head) { head = node; return; }
        Node* curr = head;
        while (curr->next) curr = curr->next;
        curr->next = node;
    }

    bool remove(int val) {
        if (!head) return false;
        if (head->data == val) {
            Node* temp = head;
            head = head->next;
            delete temp;
            return true;
        }
        Node* curr = head;
        while (curr->next && curr->next->data != val)
            curr = curr->next;
        if (!curr->next) return false;
        Node* temp = curr->next;
        curr->next = temp->next;
        delete temp;
        return true;
    }

    bool search(int val) const {
        Node* curr = head;
        while (curr) {
            if (curr->data == val) return true;
            curr = curr->next;
        }
        return false;
    }

    void display() const {
        Node* curr = head;
        while (curr) {
            cout << curr->data;
            if (curr->next) cout << " -> ";
            curr = curr->next;
        }
        cout << " -> NULL" << endl;
    }
};

int main() {
    LinkedList list;
    list.insertBack(10);
    list.insertBack(20);
    list.insertBack(30);
    list.insertFront(5);

    cout << "List: ";
    list.display();          // 5 -> 10 -> 20 -> 30 -> NULL

    cout << "Search 20: "
         << (list.search(20) ? "Found" : "Not found") << endl;

    list.remove(20);
    cout << "After removing 20: ";
    list.display();          // 5 -> 10 -> 30 -> NULL

    return 0;
}

Challenge 5: Simple Stack Implementation

Implement a stack data structure from scratch using a dynamic array (vector). Support push, pop, top, isEmpty, and size.

#include <iostream>
#include <vector>
#include <stdexcept>
using namespace std;

template <typename T>
class Stack {
    vector<T> data;
public:
    void push(const T& val) {
        data.push_back(val);
    }

    void pop() {
        if (data.empty())
            throw underflow_error("Stack underflow");
        data.pop_back();
    }

    const T& top() const {
        if (data.empty())
            throw underflow_error("Stack is empty");
        return data.back();
    }

    bool isEmpty() const { return data.empty(); }
    size_t size() const  { return data.size(); }

    void display() const {
        cout << "Stack (top -> bottom): ";
        for (int i = data.size() - 1; i >= 0; i--) {
            cout << data[i];
            if (i > 0) cout << ", ";
        }
        cout << endl;
    }
};

int main() {
    Stack<int> stack;

    stack.push(10);
    stack.push(20);
    stack.push(30);
    stack.display();           // 30, 20, 10

    cout << "Top: " << stack.top() << endl;       // 30
    cout << "Size: " << stack.size() << endl;     // 3

    stack.pop();
    cout << "After pop, top: " << stack.top() << endl; // 20
    stack.display();           // 20, 10

    cout << "Empty? "
         << (stack.isEmpty() ? "Yes" : "No") << endl; // No

    // Test with strings
    Stack<string> strStack;
    strStack.push("Hello");
    strStack.push("World");
    strStack.display();        // World, Hello

    return 0;
}

What’s Next?

Completing this course is a major milestone, but your C++ journey is far from over. Here are some paths you can explore to deepen your skills:

  • Data Structures & Algorithms – Study arrays, linked lists, trees, graphs, sorting, searching, and dynamic programming. Practise on platforms like LeetCode, HackerRank, and Codeforces.
  • Game Development – Learn libraries such as SFML, SDL2, or Unreal Engine (which uses C++ extensively). Build 2D and 3D games to apply OOP, graphics, and physics concepts.
  • Systems Programming – Explore operating system concepts, network programming with sockets, and embedded systems. C++ is a dominant language in these domains.
  • Competitive Programming – Participate in coding contests (Codeforces, AtCoder, USACO). This sharpens your problem-solving speed and algorithmic thinking.
  • Open Source Contributions – Find a C++ open-source project on GitHub that interests you. Start with “good first issue” labels and learn from real-world codebases.
  • Advanced C++ – Dive into metaprogramming, SFINAE, concepts (C++20), coroutines, modules, and ranges. Read books like Effective Modern C++ by Scott Meyers.
  • Build Your Own Projects – The best way to learn is to build something you care about. A chat application, a file compressor, a ray tracer, a database engine – the possibilities are endless.

Remember: consistency beats intensity. Code a little every day, and you will be amazed at how quickly you improve.
